* Emacs Setup up!
  Press Down Arrow!!!
*** Install ox-revel
    #+BEGIN_SRC elisp
      (use-package ox-reveal
	:ensure t
	:config
	(progn
	  (setq org-reveal-root "http://cdn.jsdelivr.net/reveal.js/3.0.0/")
	  (setq org-reveal-mathjax t)))
    #+END_SRC
*** Install htmlize
    #+BEGIN_SRC elisp
      (use-package htmlize
	:ensure t)
    #+END_SRC
